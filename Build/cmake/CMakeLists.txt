PROJECT(Air)

CMAKE_MINIMUM_REQUIRED(VERSION 3.4)


IF(COMMAND cmake_policy)
    CMAKE_POLICY(SET CMP0003 NEW)
ENDIF(COMMAND cmake_policy)


FIND_PATH(AIR_PROJECT_DIR ICON.png PATHS ../../)
MARK_AS_ADVANCED(AIR_PROJECT_DIR)
SET(AIR_ROOT_DIR "${AIR_PROJECT_DIR}")

INCLUDE(${AIR_ROOT_DIR}/cmake/Common.cmake)
INCLUDE(${AIR_ROOT_DIR}/cmake/Platform.cmake)

SET_PROPERTY(GLOBAL PROPERTY USE_FOLDERS ON)

SET(AIR_BINARY_DIR "${AIR_ROOT_DIR}/bin")
SET(AIR_OUTPUT_DIR "${AIR_BINARY_DIR}/${AIR_ARCH_NAME}/${CMAKE_BUILD_TYPE}")

ADD_CUSTOM_TARGET(allTargets)

SET_TARGET_PROPERTIES(allTargets PROPERTIES FOLDER "Engine")

ADD_SUBDIRECTORY(Engine)
